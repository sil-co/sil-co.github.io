## 3. 制御構文

プログラムはさまざまな処理を組み合わせて実行します。ここでは、Javaでの基本的な制御構文を学びます。条件分岐や繰り返し処理を理解し、プログラムの流れをコントロールしましょう。次にforループとwhileループで具体的な繰り返しの実装方法を紹介します。

3.1 if文

if文の構文

if (条件) {
    // 条件がtrueのときに実行されるコード
}

int x = 10;

if (x > 5) {
    System.out.println("xは5より大きいです。");
}

if-else文

if (条件) {
    // 条件がtrueのときに実行されるコード
} else {
    // 条件がfalseのときに実行されるコード
}

int x = 3;

if (x > 5) {
    System.out.println("xは5より大きいです。");
} else {
    System.out.println("xは5以下です。");
}

if-else if-else文

if (条件1) {
    // 条件1がtrueのときに実行されるコード
} else if (条件2) {
    // 条件1がfalseで条件2がtrueのときに実行されるコード
} else {
    // どちらの条件もfalseのときに実行されるコード
}

int score = 85;

if (score >= 90) {
    System.out.println("評価: A");
} else if (score >= 75) {
    System.out.println("評価: B");
} else {
    System.out.println("評価: C");
}

if文の入れ子（ネスト）

int num = 10;

if (num > 5) {
    System.out.println("numは5より大きいです。");
    
    if (num > 8) {
        System.out.println("numは8より大きいです。");
    }
}

import java.util.ArrayList;

public class ListExample {
    public static void main(String[] args) {
        ArrayList<Integer> numbers = new ArrayList<>();

        numbers.add(10);
        numbers.add(20);
        numbers.add(30);

        // 要素の削除
        numbers.remove(1);  // インデックス1の要素を削除

        // リストのサイズ
        System.out.println("リストのサイズ: " + numbers.size());  // 出力: 2

        // 全要素の表示
        for (Integer number : numbers) {
            System.out.println(number);
        }
    }
}

ストリームの作成

import java.util.Arrays;
import java.util.List;
import java.util.stream.Stream;

public class StreamExample {
    public static void main(String[] args) {
        List<String> names = Arrays.asList("Alice", "Bob", "Charlie");
        
        // リストからストリームを作成
        Stream<String> nameStream = names.stream();
        
        // ストリームの要素を一つずつ出力
        nameStream.forEach(System.out::println);
    }
}

ストリームの中間操作と終端操作

import java.util.Arrays;
import java.util.List;

public class StreamOperations {
    public static void main(String[] args) {
        List<String> names = Arrays.asList("Alice", "Bob", "Charlie", "David");

        names.stream()
             .filter(name -> name.startsWith("C")) // 'C'で始まる名前をフィルタ
             .map(String::toUpperCase)             // 大文字に変換
             .forEach(System.out::println);        // 結果を出力
    }
}

集約操作

import java.util.Arrays;
import java.util.List;

public class StreamReduction {
    public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(1, 2, 3, 4, 5);

        int sum = numbers.stream()
                         .reduce(0, (subtotal, element) -> subtotal + element);

        System.out.println("合計: " + sum);
    }
}
