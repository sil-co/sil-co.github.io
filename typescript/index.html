型推論

型推論の基本

let message = "Hello, TypeScript!";

message = 42; // エラー: Type 'number' is not assignable to type 'string'.

型推論の具体例

let num = 10; // numはnumber型と推論される。
let name = "Alice"; // nameはstring型と推論される。
let isActive = true; // isActiveはboolean型と推論される。

let numbers = [1, 2, 3]; // 数値の配列としてnumber[]型が推論される。
let person = { name: "John", age: 25 }; // 人を表すオブジェクトとして型が推論される。

function add(a: number, b: number) {
    return a + b; // 戻り値はnumber型と推論される。
}

let result = add(5, 10); // resultはnumber型と推論される。

function identity<T>(arg: T): T {
    return arg; // argの型がTとして推論され、戻り値もTとなる。
}

let str = identity("Hello"); // strはstring型と推論される。
let numIdentity = identity(42); // numIdentityはnumber型と推論される。

クラスと継承

クラスの定義

基本的なクラスの定義

class Person {
    // プロパティ
    name: string;
    age: number;

    // コンストラクタ
    constructor(name: string, age: number) {
        this.name = name;
        this.age = age;
    }

    // メソッド
    greet(): string {
        return `こんにちは、私の名前は${this.name}で、${this.age}歳です。`;
    }
}

// クラスのインスタンス化
const person1 = new Person('太郎', 25);
console.log(person1.greet()); // 出力: こんにちは、私の名前は太郎で、25歳です。

アクセス修飾子

class Person {
    public name: string;
    private age: number;

    constructor(name: string, age: number) {
        this.name = name;
        this.age = age;
    }

    public greet(): string {
        return `こんにちは、私の名前は${this.name}で、年齢は非公開です。`;
    }
}

const person1 = new Person('太郎', 25);
console.log(person1.greet()); // 出力: こんにちは、私の名前は太郎で、年齢は非公開です。
// console.log(person1.age); // エラー: 'age'はプライベートで、'Person'クラス内でのみアクセス可能

継承

class Employee extends Person {
    private position: string;

    constructor(name: string, age: number, position: string) {
        super(name, age); // 基底クラスのコンストラクタを呼び出す
        this.position = position;
    }

    public describe(): string {
        return `${this.name}は${this.position}として働いています。`;
    }
}

const employee1 = new Employee('花子', 30, 'エンジニア');
console.log(employee1.greet()); // 出力: こんにちは、私の名前は花子で、年齢は非公開です。
console.log(employee1.describe()); // 出力: 花子はエンジニアとして働いています。

        this.balance += amount;
    }

        return this.balance;

const account = new BankAccount(1000);

console.log(account.getBalance()); // 1500
// console.log(account.balance); // エラー: 'balance'はprivateです。

}

継承

class Animal {

    speak() {
        console.log(`${this.name}が鳴いています。`);
    }

    speak() {
        console.log(`${this.name}がワンワンと鳴いています。`);

ポリモーフィズム

interface Animal {
    speak(): void;
}

class Cat implements Animal {
    constructor(public name: string) {}

    speak() {
        console.log(`${this.name}がニャーと鳴いています。`);
    }
}

class Dog implements Animal {
    constructor(public name: string) {}

    speak() {
        console.log(`${this.name}がワンワンと鳴いています。`);
    }
}

function makeAnimalSpeak(animal: Animal) {
    animal.speak();
}

const cat = new Cat('ミケ');
const dog = new Dog('ポチ');

makeAnimalSpeak(cat); // ミケがニャーと鳴いています。
makeAnimalSpeak(dog); // ポチがワンワンと鳴いています。

}

ジェネリクスを使用したクラス

class Box<T> {
    private contents: T;

    constructor(value: T) {
        this.contents = value;
    }

    public getContents(): T {
        return this.contents;
    }
}

const numberBox = new Box<number>(123);
console.log(numberBox.getContents()); // 123

const stringBox = new Box<string>("Hello!");
console.log(stringBox.getContents()); // Hello!

function logProperty(target: any, propertyKey: string) {
    let value = target[propertyKey];

    const getter = () => {
        console.log(`取得: ${propertyKey} => ${value}`);
        return value;
    };

    const setter = (newValue: any) => {
        console.log(`設定: ${propertyKey} => ${newValue}`);
        value = newValue;
    };

    // プロパティを置き換える
    Object.defineProperty(target, propertyKey, {
        get: getter,
        set: setter,
        enumerable: true,
        configurable: true,
    });
}

class MyClass {

    public myProp: string;

    constructor() {
        this.myProp = '初期値';

}

// src/server.ts
let users: { id: number, name: string }[] = [];
let idCounter = 1;

app.get('/users', (req: Request, res: Response) => {
    res.json(users);
});

app.post('/users', (req: Request, res: Response) => {
    const newUser = { id: idCounter++, name: req.body.name };
    users.push(newUser);
    res.status(201).json(newUser);
});

app.put('/users/:id', (req: Request, res: Response) => {
    const userId = parseInt(req.params.id);
    const user = users.find(u => u.id === userId);

    if (user) {
        user.name = req.body.name;
        res.json(user);
    } else {
        res.status(404).send('User not found');
    }
});

app.delete('/users/:id', (req: Request, res: Response) => {
    users = users.filter(u => u.id !== parseInt(req.params.id));
    res.status(204).send();
});

オブザーバーパターン

interface Observer {
    update(data: string): void;
}

class Subject {
    private observers: Observer[] = [];

    public addObserver(observer: Observer): void {
        this.observers.push(observer);
    }

    public removeObserver(observer: Observer): void {
        this.observers = this.observers.filter(obs => obs !== observer);
    }

    public notifyObservers(data: string): void {
        this.observers.forEach(observer => observer.update(data));
    }
}

class ConcreteObserver implements Observer {
    private name: string;

    constructor(name: string) {
        this.name = name;
    }

    update(data: string): void {
        console.log(`${this.name} received: ${data}`);
    }
}

// 使用例
const subject = new Subject();
const observer1 = new ConcreteObserver("Observer 1");
const observer2 = new ConcreteObserver("Observer 2");

subject.addObserver(observer1);
subject.addObserver(observer2);
subject.notifyObservers("Hello, Observers!");

// Output:
// Observer 1 received: Hello, Observers!
// Observer 2 received: Hello, Observers!

ストラテジーパターン

interface Strategy {
    execute(a: number, b: number): number;
}

class AdditionStrategy implements Strategy {
    execute(a: number, b: number): number {
        return a + b;
    }
}

class SubtractionStrategy implements Strategy {
    execute(a: number, b: number): number {
        return a - b;
    }
}

class Context {
    private strategy: Strategy;

    constructor(strategy: Strategy) {
        this.strategy = strategy;
    }

    setStrategy(strategy: Strategy): void {
        this.strategy = strategy;
    }

    calculate(a: number, b: number): number {
        return this.strategy.execute(a, b);
    }
}

// 使用例
const context = new Context(new AdditionStrategy());
console.log(context.calculate(5, 3)); // 8

context.setStrategy(new SubtractionStrategy());
console.log(context.calculate(5, 3)); // 2
